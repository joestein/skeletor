with placement_strategy = 'org.apache.cassandra.locator.SimpleStrategy'
and strategy_options = [{replication_factor:1}];

WITH comparator = UTF8Type
AND key_validation_class=UTF8Type;

WITH comparator = UTF8Type
AND key_validation_class=UTF8Type;
